print(FtpMessageEncoder().encode(ftpMessage))

print("Encode Employee Object into JSON formatted Data using custom JSONEncoder")
ftpJSONData = json.dumps(ftpMessage, indent=4, cls=FtpMessageEncoder)

print(ftpJSONData)

print("Decode JSON formatted Data")
ftpMessageDecoded = json.loads(ftpJSONData)
print(ftpMessageDecoded)

print(ftpMessageDecoded['session_id'])
